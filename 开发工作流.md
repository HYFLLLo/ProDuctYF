```python
import json
import hashlib
from datetime import datetime
from abc import ABC, abstractmethod

class BaseCollector(ABC):
    def __init__(self, source_name: str, api_config: dict):
        self.source_name = source_name
        self.api_config = api_config

    @abstractmethod
    async def collect(self, time_range: tuple[datetime, datetime]) -> list[dict]:
        pass

    async def normalize(self, raw_event: dict) -> dict:
        return {
            "event_id": self._generate_event_id(raw_event),
            "source": self.source_name,
            "raw_content": json.dumps(raw_event, ensure_ascii=False),
            "collected_at": datetime.now(),
            "status": "pending"
        }

    def _generate_event_id(self, raw_event: dict) -> str:
        content = f"{raw_event.get('name', '')}{raw_event.get('time', '')}"
        return hashlib.md5(content.encode()).hexdigest()[:16]
```

```python
import httpx
from datetime import datetime
from .base import BaseCollector

class WeatherCollector(BaseCollector):
    async def collect(self, time_range: tuple[datetime, datetime]) -> list[dict]:
        async with httpx.AsyncClient() as client:
            response = await client.get(
                self.api_config["url"],
                params={
                    "key": self.api_config["api_key"],
                    "location": self.api_config.get("location", "beijing")
                }
            )
            data = response.json()
            return self._parse_weather_data(data)

    def _parse_weather_data(self, data: dict) -> list[dict]:
        events = []
        for item in data.get("HeWeather", [{}])[0].get("hourly", []):
            if self._is_night_hours(item["time"]):
                events.append({
                    "name": f"天气：{item['text']}",
                    "summary": f"温度{item['temp']}度，{item['text']}，风力{item['windDir']}",
                    "time": item["time"],
                    "location": self.api_config.get("location")
                })
        return events

    def _is_night_hours(self, time_str: str) -> bool:
        hour = int(time_str.split()[1].split(":")[0])
        return hour >= 20 or hour <= 6
```

```python
import json
from datetime import datetime
from typing import Optional

CLASSIFICATION_PROMPT = """
你是一个事件分类专家。请根据以下事件信息，判断其类型。

事件信息：
- 事件名称：{event_name}
- 事件摘要：{event_summary}
- 发生时间：{event_time}

分类选项：
- 赛事：体育比赛相关事件
- 娱乐：影视、音乐、综艺等娱乐事件
- 天气：天气相关事件
- 社会：社会新闻、热点事件
- 其他：不属于以上类别的事件

请输出JSON格式的分类结果：
{{
    "category": "分类名称",
    "confidence": 0.0-1.0的置信度,
    "reasoning": "分类理由"
}}
"""

class EventClassifier:
    def __init__(self, llm_client):
        self.llm = llm_client

    async def classify(self, event: dict) -> dict:
        prompt = CLASSIFICATION_PROMPT.format(
            event_name=event.get("name", ""),
            event_summary=event.get("summary", ""),
            event_time=event.get("time", "")
        )
        response = await self.llm.agenerate([prompt])
        result = self._parse_response(response)

        if result["confidence"] < 0.7:
            result["needs_review"] = True

        return result

    def _parse_response(self, response) -> dict:
        text = response.generations[0].text.strip()
        try:
            return json.loads(text)
        except:
            return {"category": "其他", "confidence": 0.5, "reasoning": "解析失败"}
```

### Step 6：开发TF-IDF去重模块

**目标**：实现语义去重引擎

**创建文件**：`d:\外卖夜间爆品预测\src\services\event_dedup.py`

```python
import numpy as np
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity
from typing import Optional

class EventDedup:
    def __init__(self, similarity_threshold: float = 0.85):
        self.similarity_threshold = similarity_threshold
        self.vectorizer = TfidfVectorizer()
        self.event_index: list[dict] = []
        self.vector_index: Optional[np.ndarray] = None

    async def process(self, events: list[dict]) -> list[dict]:
        if not events:
            return []

        unique_events = []
        for event in events:
            if not await self._is_duplicate(event):
                unique_events.append(event)
                await self._add_to_index(event)

        return unique_events

    async def _is_duplicate(self, event: dict) -> bool:
        if not self.event_index:
            return False

        event_text = self._get_event_text(event)
        event_vector = self.vectorizer.transform([event_text])
        similarities = cosine_similarity(event_vector, self.vector_index)[0]
        max_similarity = float(np.max(similarities))

        return max_similarity >= self.similarity_threshold

    async def _add_to_index(self, event: dict):
        event_text = self._get_event_text(event)
        new_vector = self.vectorizer.fit_transform([event_text])

        if self.vector_index is None:
            self.vector_index = new_vector.toarray()
        else:
            self.vector_index = np.vstack([self.vector_index, new_vector.toarray()])

        self.event_index.append(event)

    def _get_event_text(self, event: dict) -> str:
        return f"{event.get('name', '')} {event.get('summary', '')}"
```

### Step 7：开发热度计算模块

**目标**：实现热度计算和衰减

**创建文件**：`d:\外卖夜间爆品预测\src\services\heat_calculator.py`

```python
import math
from datetime import datetime
from typing import Optional

class HeatCalculator:
    WEIGHTS = {"view_count": 0.4, "click_count": 0.6}
    DECAY_CONFIG = {"half_life_hours": 6, "base_decay": 0.95}

    def __init__(self, redis_client=None):
        self.redis = redis_client

    def calculate(self, event: dict) -> float:
        view_count = event.get("view_count", 0)
        click_count = event.get("click_count", 0)

        raw_heat = (
            view_count * self.WEIGHTS["view_count"] +
            click_count * self.WEIGHTS["click_count"]
        )

        normalized_heat = self._normalize(raw_heat)
        age_hours = self._get_event_age_hours(event)
        decayed_heat = self._apply_decay(normalized_heat, age_hours)

        return round(decayed_heat, 2)

    def _normalize(self, raw_heat: float) -> float:
        log_heat = math.log1p(raw_heat)
        max_log = math.log1p(1000000)
        return min(100, (log_heat / max_log) * 100)

    def _apply_decay(self, heat: float, age_hours: float) -> float:
        decay_factor = pow(
            self.DECAY_CONFIG["base_decay"],
            age_hours / self.DECAY_CONFIG["half_life_hours"]
        )
        return heat * decay_factor

    def _get_event_age_hours(self, event: dict) -> float:
        created_at = event.get("created_at", datetime.now())
        if isinstance(created_at, str):
            created_at = datetime.fromisoformat(created_at)
        age = datetime.now() - created_at
        return age.total_seconds() / 3600
```

**创建文件**：`d:\外卖夜间爆品预测\src\services\rule_engine.py`

```python
from datetime import datetime
from typing import Optional

SCENE_RULES = {
    "看球": {
        "conditions": [
            {"type": "order_contains", "category": "啤酒"},
            {"type": "order_contains", "category": "零食"},
            {"type": "event_active", "event_type": "赛事"},
            {"type": "time_range", "start": "20:00", "end": "06:00"}
        ],
        "condition_logic": "AND",
        "weight": 0.8
    },
    "加班": {
        "conditions": [
            {"type": "order_contains", "category": "泡面"},
            {"type": "order_contains", "category": "能量饮料"},
            {"type": "location_type", "value": "写字楼"},
            {"type": "is_workday"},
            {"type": "time_range", "start": "20:00", "end": "23:00"}
        ],
        "condition_logic": "AND",
        "weight": 0.75
    },
    "聚会": {
        "conditions": [
            {"type": "order_count", "operator": ">=", "value": 3},
            {"type": "order_contains", "category": "酒水", "count": ">=", 2},
            {"type": "is_weekend_or_holiday"},
            {"type": "order_total_amount", "operator": ">=", "value": 150}
        ],
        "condition_logic": "AND",
        "weight": 0.85
    }
}

class RuleEngine:
    def __init__(self):
        self.rules = SCENE_RULES

    async def match(self, user_order: dict, user_profile: dict, event_context: dict) -> list[dict]:
        candidates = []

        for scene_name, rule in self.rules.items():
            match_result = await self._evaluate_rule(rule, user_order, user_profile, event_context)
            if match_result["matched"]:
                candidates.append({
                    "scene": scene_name,
                    "confidence": match_result["confidence"],
                    "matched_conditions": match_result["matched_conditions"],
                    "weight": rule["weight"]
                })

        candidates.sort(key=lambda x: x["confidence"] * x["weight"], reverse=True)
        return candidates[:3]

    async def _evaluate_rule(self, rule: dict, user_order: dict, user_profile: dict, event_context: dict) -> dict:
        matched_conditions = []
        total_confidence = 0

        for condition in rule["conditions"]:
            result = await self._check_condition(condition, user_order, user_profile, event_context)
            if result["matched"]:
                matched_conditions.append(condition)
                total_confidence += result.get("confidence", 0.8)

        if rule["condition_logic"] == "AND":
            matched = len(matched_conditions) == len(rule["conditions"])
        else:
            matched = len(matched_conditions) > 0

        confidence = total_confidence / len(rule["conditions"]) if matched else 0

        return {
            "matched": matched,
            "confidence": confidence,
            "matched_conditions": matched_conditions
        }

    async def _check_condition(self, condition: dict, user_order: dict, user_profile: dict, event_context: dict) -> dict:
        cond_type = condition["type"]

        if cond_type == "order_contains":
            categories = user_order.get("categories", [])
            return {"matched": condition["category"] in categories, "confidence": 0.9}

        elif cond_type == "time_range":
            current_hour = datetime.now().hour
            start_hour = int(condition["start"].split(":")[0])
            end_hour = int(condition["end"].split(":")[0])
            matched = start_hour <= current_hour or current_hour <= end_hour
            return {"matched": matched, "confidence": 0.7}

        elif cond_type == "is_workday":
            is_workday = datetime.now().weekday() < 5
            return {"matched": is_workday, "confidence": 0.8}

        elif cond_type == "event_active":
            active_events = event_context.get("active_events", [])
            event_types = [e.get("type") for e in active_events]
            return {"matched": condition["event_type"] in event_types, "confidence": 0.85}

        return {"matched": False, "confidence": 0}
```

```python
import pandas as pd
from datetime import datetime, timedelta
from prophet import Prophet

class SalesPredictor:
    def __init__(self, model_path: str = None):
        self.model_path = model_path or "./models/sales_predictor"
        self.model = self._load_model()

    def _load_model(self):
        model = Prophet(
            yearly_seasonality=False,
            weekly_seasonality=True,
            daily_seasonality=True,
            seasonality_mode='multiplicative',
            growth='linear'
        )
        model.add_seasonality(
            name='night_time',
            period=6,
            fourier_order=3
        )
        return model

    async def predict(
        self,
        merchant_id: str,
        product_id: str,
        historical_sales: list[dict],
        prediction_hours: int = 6
    ) -> dict:
        df = self._prepare_historical_df(historical_sales)

        if len(df) > 0:
            self.model.fit(df)

        future_df = self._make_future_hours_df(prediction_hours)
        forecast = self.model.predict(future_df)

        hourly_predictions = self._extract_hourly_predictions(forecast, prediction_hours)

        return {
            "product_id": product_id,
            "predicted_hourly_avg": float(forecast['yhat'].mean()),
            "predicted_hourly": hourly_predictions,
            "confidence_lower": float(forecast['yhat_lower'].mean()),
            "confidence_upper": float(forecast['yhat_upper'].mean())
        }

    def _prepare_historical_df(self, sales_data: list[dict]) -> pd.DataFrame:
        if not sales_data:
            return pd.DataFrame(columns=['ds', 'y'])

        df = pd.DataFrame(sales_data)
        df['ds'] = pd.to_datetime(df['order_time'])
        df['y'] = df['quantity']
        return df[['ds', 'y']]

    def _make_future_hours_df(self, hours: int) -> pd.DataFrame:
        now = datetime.now()
        dates = [now + timedelta(hours=i) for i in range(hours)]
        return pd.DataFrame({'ds': dates})

    def _extract_hourly_predictions(self, forecast, hours: int) -> dict[int, float]:
        hourly = {}
        for i, row in forecast.iterrows():
            if i < hours:
                hourly[i] = float(row['yhat'])
        return hourly
```

**创建文件**：`d:\外卖夜间爆品预测\src\services\restock_calculator.py`

```python
from typing import Optional

class RestockCalculator:
    def __init__(self, safety_stock_days: float = 1.5):
        self.safety_stock_days = safety_stock_days

    async def calculate(
        self,
        merchant_id: str,
        sales_predictions: dict[str, dict],
        inventory_data: dict[str, dict],
        replenishment_period_hours: int = 24
    ) -> list[dict]:
        recommendations = []

        for product_id, prediction in sales_predictions.items():
            inv = inventory_data.get(product_id, {})
            current_stock = inv.get("usable_stock", 0)

            predicted_hourly_avg = prediction.get("predicted_hourly_avg", 0)
            predicted_demand = predicted_hourly_avg * replenishment_period_hours

            safety_stock = self._calculate_safety_stock(predicted_hourly_avg)
            recommended_qty = max(0, predicted_demand - current_stock + safety_stock)
            urgency = self._determine_urgency(current_stock, safety_stock, predicted_demand)

            recommendations.append({
                "product_id": product_id,
                "current_stock": current_stock,
                "safety_stock": round(safety_stock, 2),
                "predicted_demand": round(predicted_demand, 2),
                "recommended_quantity": max(1, int(recommended_qty)),
                "urgency": urgency
            })

        urgency_order = {"critical": 0, "high": 1, "medium": 2, "low": 3}
        recommendations.sort(key=lambda x: (urgency_order[x["urgency"]], -x["recommended_quantity"]))

        return recommendations

    def _calculate_safety_stock(self, predicted_hourly_avg: float) -> float:
        return predicted_hourly_avg * self.safety_stock_days

    def _determine_urgency(
        self,
        current_stock: float,
        safety_stock: float,
        predicted_demand: float
    ) -> str:
        if predicted_demand <= 0:
            return "low"

        coverage_ratio = current_stock / predicted_demand

        if current_stock < safety_stock * 0.5 or coverage_ratio < 0.5:
            return "critical"
        elif current_stock < safety_stock or coverage_ratio < 1.0:
            return "high"
        elif coverage_ratio < 1.5:
            return "medium"
        return "low"
```

```python
class LLMCircuitBreaker:
    def __init__(self, failure_threshold: int = 5, recovery_timeout: int = 60):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.failure_count = 0
        self.last_failure_time: Optional[float] = None
        self.state = "closed"

    def record_success(self):
        self.failure_count = 0
        self.state = "closed"

    def record_failure(self):
        self.failure_count += 1
        self.last_failure_time = time.time()

        if self.failure_count >= self.failure_threshold:
            self.state = "open"

    def is_open(self) -> bool:
        if self.state == "open":
            if time.time() - self.last_failure_time >= self.recovery_timeout:
                self.state = "half-open"
                return False
            return True
        return False


class LLMWithFallback:
    def __init__(self, primary_model: str, fallback_models: list[str], redis_client=None):
        self.primary_model = primary_model
        self.fallback_models = fallback_models
        self.redis = redis_client
        self.circuit_breaker = LLMCircuitBreaker()

    async def generate(self, prompt: str, temperature: float = 0.7, max_tokens: int = 1000) -> str:
        cache_key = self._get_cache_key(prompt)
        cached = await self._get_from_cache(cache_key)
        if cached:
            return cached

        if self.circuit_breaker.is_open():
            return await self._call_fallback(prompt)

        try:
            result = await self._call_primary(prompt, temperature, max_tokens)
            self.circuit_breaker.record_success()
            await self._save_to_cache(cache_key, result)
            return result
        except Exception as e:
            self.circuit_breaker.record_failure()
            return await self._call_fallback(prompt)

    async def _call_primary(self, prompt: str, temperature: float, max_tokens: int) -> str:
        pass

    async def _call_fallback(self, prompt: str) -> str:
        for model in self.fallback_models:
            try:
                result = await self._call_model(model, prompt)
                return result
            except:
                continue
        raise Exception("All fallback models failed")

    async def _call_model(self, model: str, prompt: str) -> str:
        pass

    def _get_cache_key(self, prompt: str) -> str:
        return hashlib.md5(prompt.encode()).hexdigest()

    async def _get_from_cache(self, key: str) -> Optional[str]:
        if self.redis:
            return await self.redis.get(f"llm:cache:{key}")

    async def _save_to_cache(self, key: str, value: str, ttl: int = 3600):
        if self.redis:
            await self.redis.setex(f"llm:cache:{key}", ttl, value)
```
